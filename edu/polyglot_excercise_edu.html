Simple Ops — C++

#include <iostream>
int main(){
  int a = 7; double b = 3.5;
  std::cout << "sum: " << (a + b) << "\n";
  return 0;
}

Simple Ops — Java

public class App{
  public static void main(String[]args){
    int a=7; double b=3.5;
    System.out.println("sum: " + (a + b));
  }
}

Simple Ops — JavaScript

const a = 7, b = 3.5;
console.log(`sum: ${a + b}`);

Simple Ops — Swift

let a = 7
let b = 3.5
print("sum: \(Double(a) + b)")

Simple Ops — Objective‑C

#import <Foundation/Foundation.h>
int main(){ @autoreleasepool{
  int a = 7; double b = 3.5;
  NSLog(@"sum: %f", a + b);
  return 0; }
}

Simple Ops — Python

a = 7
b = 3.5
print(f"sum: {a + b}")

Simple Ops — C

#include <stdio.h>
int main(){
  int a=7; double b=3.5;
  printf("sum: %g\n", a + b);
  return 0;
}

Conditionals — C++

int x = 10;
if(x > 5) std::cout << "big\n";
else std::cout << "small\n";

Conditionals — Java

int x = 10;
if(x > 5) System.out.println("big");
else System.out.println("small");

Conditionals — JavaScript

const x = 10;
console.log(x > 5 ? 'big' : 'small');

Conditionals — Swift

let x = 10
print(x > 5 ? "big" : "small")

Conditionals — Objective‑C

int x = 10;
NSLog(@"%@", (x > 5) ? @"big" : @"small");

Conditionals — Python

x = 10
print('big' if x > 5 else 'small')

Conditionals — C

int x = 10;
printf("%s\n", (x > 5) ? "big" : "small");

for i in 0..<5 { print(i) }

Loops — Objective‑C

Loops — Python

for i in range(5):
  print(i)

Loops — C

Classes — C++

};
